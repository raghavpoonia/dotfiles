# requirements-ml.txt
# raghavpoonia/dotfiles
#
# AI/ML Python dependencies
# Install via: pip install -r requirements-ml.txt
# Or via:      bash bootstrap.sh --ml
#
# When do you need this?
#   - Training or fine-tuning ML models
#   - Running LLM inference locally
#   - Working with transformers, embeddings, or vector stores
#   - Running or converting Jupyter notebooks
#   - Building AI security tools (model scanning, prompt injection detection)
#
# When you DON'T need this?
#   - General Python scripting
#   - Security tooling without ML components
#   - Web development
#   - DevSecOps automation
#
# GPU notes:
#   M-series Mac  torch uses Metal/MPS automatically — no extra config
#   Intel Mac     CPU only — torch works but training is slow
#   Check: python3 -c "import torch; print(torch.backends.mps.is_available())"


# ── Core data science ─────────────────────────────────────────────────────────
numpy
pandas
scikit-learn
matplotlib
seaborn
scipy

# ── Deep learning ─────────────────────────────────────────────────────────────
# torch — Metal/MPS on M-series, CPU on Intel (same package, auto-detects)
torch

torchaudio

datasets
